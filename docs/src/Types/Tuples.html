rule fst-pair
* \fst(\pair(f)(g)(t)) == f(t)

rule snd-pair
* \snd(\pair(f)(g)(t)) == g(t)

rule pair-unique

  * \fst(h(t)) == f(t)
  * \snd(h(t)) == g(t)

  * h == \pair
