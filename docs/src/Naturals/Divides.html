type \div :: Nat -> Nat -> Bool

definition def-div
* \div(a)(b) == \eq(\zero)(\rem(b)(a))
theorem div-zero
* \div(a)(\zero) == \true

proof
1. \div(a)(\zero) : chain
    == \eq(\zero)(\rem(\zero)(a))
     : use def-div;
    == \eq(\zero)(\snd(\divalg(\zero)(a)))
     : use def-rem; at z in
       \eq(\zero)(z)
    == \eq(\zero)(\snd(\tup(\zero)(\zero)))
     : use divalg-zero-l; at z in
       \eq(\zero)(\snd(z))
    == \eq(\zero)(\zero)
     : use snd-tup; at z in
       \eq(\zero)(z)
    == \true
     : use eq-refl;
theorem zero-div
if
  * \div(\zero)(a) == \true
then
  * a == \zero

proof
1. \true : chain
    == \div(\zero)(a)
     : flop assumption 1
    == \eq(\zero)(\rem(a)(\zero))
     : use def-div;
    == \eq(\zero)(\snd(\divalg(a)(\zero)))
     : use def-rem; at z in
       \eq(\zero)(z)
    == \eq(\zero)(\snd(\tup(\zero)(a)))
     : use divalg-zero-r; at z in
       \eq(\zero)(\snd(z))
    == \eq(\zero)(a)
     : use snd-tup; at z in
       \eq(\zero)(z)
    == \eq(a)(\zero)
     : use eq-comm;

2. \eq(a)(\zero) : chain
    == \true
     : flop use reiterate; 1

3. a == \zero
    : use eq-dereify; 2
theorem div-impl-times

  * ∃k. b == \times(a)(k)
