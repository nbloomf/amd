Identity and Composition

type \id :: a -> a

rule def-id
* \id(x) == x
rule fun-eq
if
  * ∀x. f(x) == g(x)
then
  * f == g
rule fun-ap
if
  * f == g
then
  * f(x) == g(x)
theorem fun-ap-2
if
  * f == g
then
  * f(x)(y) == g(x)(y)

proof
1. f == g : assumption 1
2. f(x) == g(x) : use fun-ap; 1
3. f(x)(y) == g(x)(y) : use fun-ap; 2
type \comp :: (b -> c) -> (a -> b) -> a -> c

rule def-comp
* \comp(g)(f)(a) == g(f(a))
theorem comp-id-l
* \comp(\id)(f) == f

proof
1. \comp(\id)(f)(a) : chain
    == \id(f(a)) : use def-comp;
    == f(a) : use def-id;
2. ∀x. \comp(\id)(f)(x) == f(x) : forall-intro a -> x; 1
3. \comp(\id)(f) == f : use fun-eq; 2
theorem comp-id-r
* \comp(f)(\id) == f

proof
1. \comp(f)(\id)(a) : chain
    == f(\id(a)) : use def-comp;
    == f(a) : use def-id; at z in f(z)
2. ∀x. \comp(f)(\id)(x) == f(x) : forall-intro a -> x; 1
3. \comp(f)(\id) == f : use fun-eq; 2
