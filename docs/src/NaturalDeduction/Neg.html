Negation

rule neg-intro
if
  * A => B
  * A => (~B)
then
  * ~A

rule neg-elim
if
  * ~A
then
  * A => B
theorem simp
if
  * P
then
  * Q => P

proof
1. P : assumption 1
2.   Q : hypothesis hyp-Q
3.   P /\ Q : use conj-intro; 1, 2
4.   P : use conj-elim-l; 3
5. Q => P : discharge hyp-Q; 4
theorem contraposition
if
  * P => Q
then
  * (~Q) => (~P)

proof
1.   ~Q : hypothesis hyp-Q
2.   P => (~Q) : use simp; 1
3.   P => Q : assumption 1
4.   ~P : use neg-intro; 3, 2
5. (~Q) => (~P) : discharge hyp-Q; 4
